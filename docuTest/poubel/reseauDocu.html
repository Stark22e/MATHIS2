A regular reseau (=a net) 

module mathis{
    let creator=new reseau.Regular()
    /** the basis of the reseau.
     * For square net, points are i*Vi + j*Vj
     * For triangular net, points are :
     *  i*Vi + (j+decay)*Vj where decay=0.5 when j is odd*/
    creator.Vi=new XYZ(0.2,0,0)
    creator.Vj=new XYZ(0,0.2,0)
    /**number of vertices in i/j direction*/
    creator.nbI=3
    creator.nbJ=4
    creator.origine=new XYZ(-0.7,-0.7,0)
    /**true-> square net, false->triangular net*/
    creator.squareVersusTriangleMaille=true
    let mamesh=creator.go()
}

A 3d reseau 

module mathis{
    let creator = new reseau.Regular3D()
    creator.nbI = 3
    creator.nbJ = 4
    creator.nbK = 5
    creator.Vi = new XYZ(0.2, 0,   0)
    creator.Vj = new XYZ(0.2, 0.2,   0)
    creator.Vk = new XYZ(0,   0,   0.2)
    creator.origine = new XYZ(-0.7, -0.7, -0.7)

    creator.decayOddStrates=true
    creator.interStrateMailleAreSquareVersusTriangle=true
    creator.strateHaveSquareMailleVersusTriangleMaille=false

    let mamesh = creator.go()
}
